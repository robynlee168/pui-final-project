function updateGallery() { var season = document.getElementById("season-select"); var selectedValue = season.options[season.selectedIndex].value; var thumbnails = document.querySelectorAll(".spring, .summer, .fall, .winter"); var row = document.querySelector('.row'); row.style.display = "unset"; for (let i = 0; i < thumbnails.length; i++) { if ((selectedValue == "none") || (!thumbnails[i].classList.contains(selectedValue))) thumbnails[i].style.display = "none"; else thumbnails[i].style.display = "initial"; } }
